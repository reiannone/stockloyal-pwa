BACKEND Step 1: SINGLE Endpoint PHP: PowerShell Use this exact working script:
~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
# 1. Path to your PEM key
$KEY = "C:\Users\reian\.ssh\stockloyal-pwa-key-pair.pem"

# 2. Local file to upload
$SRC = "C:\xampp\htdocs\stockloyal-pwa\api\export-payments-file.php"

# 3. EC2 Host (rename variable so we avoid conflict with $Host)
$EC2 = "ec2-user@3.150.49.91"

# 4. Destination on remote server
$DEST = "/home/ec2-user/export-payments-file.php"

# 5. Upload using scp â€” THIS WORKS IN POWERSHELL
scp -i $KEY "$SRC" "${EC2}:${DEST}"

BACKEND Step 1: MULTIPLE Endpoints PHP: PowerShell Use this exact working script:
~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
$KEY = "C:\Users\reian\.ssh\stockloyal-pwa-key-pair.pem"
$EC2 = "ec2-user@3.150.49.91"
$BASE_DIR = "C:\xampp\htdocs\stockloyal-pwa\api"

# Upload files
scp -i $KEY "$BASE_DIR\get-brokers.php" "${EC2}:/home/ec2-user/"
scp -i $KEY "$BASE_DIR\save-broker.php" "${EC2}:/home/ec2-user/"
scp -i $KEY "$BASE_DIR\delete-broker.php" "${EC2}:/home/ec2-user/"

# Move to web directory and set permissions
ssh -i $KEY $EC2 "sudo mv /home/ec2-user/get-brokers.php /var/www/html/api/ && sudo mv /home/ec2-user/save-broker.php /var/www/html/api/ && sudo mv /home/ec2-user/delete-broker.php /var/www/html/api/ && sudo chown apache:apache /var/www/html/api/get-brokers.php /var/www/html/api/save-broker.php /var/www/html/api/delete-broker.php && sudo chmod 644 /var/www/html/api/get-brokers.php /var/www/html/api/save-broker.php /var/www/html/api/delete-broker.php"
